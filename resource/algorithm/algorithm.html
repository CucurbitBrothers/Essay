# 奥利给

  1. 两数之和
# 来源：力扣（LeetCode）
def twoSum(nums, target):
    hashmap={}
    for i, num in enumerate(nums):
        if hashmap.get(target - num) is not None:
            return [i, hashmap.get(target - num)]
        hashmap[num] = i #这句不能放在if语句之前，解决list中有重复值或target-num=num的情况
